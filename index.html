GOOGLE SHEETS

    
Apps Script 웹앱 URL을 입력하면 구글 시트에 데이터가 저장됩니다.

    
설정 방법 (최초 1회)
1. 📊 공개보고서 구글 시트 열기
2. 확장 프로그램 → Apps Script
3. 아래 코드 복사 → 저장 → 배포 → 새 배포
4. 웹앱 · 모든 사용자 → 배포 → URL 입력
5. 📷 카메라(OCR) 사용 시: Apps Script → 프로젝트 설정 → 스크립트 속성 → 추가
   • Gemini 사용: GEMINI_API_KEY = (Google AI Studio에서 발급)
   • Claude 사용: CLAUDE_API_KEY = (Anthropic Console에서 발급)

    
📋 Apps Script 코드 보기
      
// ── 필수 설정값 ──────────────────────────────
const SHEET_ID   = 'YOUR_SPREADSHEET_ID';  // ← 구글 시트 ID 입력
const SHEET_NAME = '전도보고서';

// 알림 수신 이메일 목록
const ADMIN_EMAILS = ['admin@example.com'];
const SEND_EMAIL = true;

// ── doPost : 보고서 저장 + OCR + 이메일 알림 ──
function doPost(e){
  try{
    const d=JSON.parse(e.postData.contents);
    if(d.action==='save_jo'){
      PropertiesService.getScriptProperties().setProperty('JO_DATA',JSON.stringify(d.joData));
      return ok({});
    }

    // ── Claude Vision OCR ─────────────────────
    if(d.action==='ocr_claude'){
      const apiKey=PropertiesService.getScriptProperties().getProperty('CLAUDE_API_KEY');
      if(!apiKey) return error('API 키가 설정되지 않았습니다. GAS 스크립트 속성에 CLAUDE_API_KEY를 등록하세요.');
      const payload={
        model:'claude-opus-4-5',
        max_tokens:1024,
        messages:[{role:'user',content:[
          {type:'image',source:{type:'base64',media_type:d.mediaType||'image/jpeg',data:d.imageBase64}},
          {type:'text',text:'이 이미지에서 전도 보고서 작성에 필요한 정보를 추출해 주세요. JSON으로만 응답하세요: {"location":"거주지역","name":"성씨","age":"나이(숫자)","marriage":"기혼또는미혼","gender":"남자또는여자","meet_place":"만난장소","job":"직업","religion":"종교배경","target_type":"대상자구분","heaven":"있었고또는없었고","reason":"믿음또는행위등","q10":"특별경험","q12":"기도제목"}'}
        ]}]
      };
      const resp=UrlFetchApp.fetch('https://api.anthropic.com/v1/messages',{
        method:'post',
        headers:{'Content-Type':'application/json','x-api-key':apiKey,'anthropic-version':'2023-06-01'},
        payload:JSON.stringify(payload),
        muteHttpExceptions:true
      });
      const res=JSON.parse(resp.getContentText());
      if(res.error) return error('Claude API 오류: '+res.error.message);
      const text=(res.content&&res.content[0]&&res.content[0].text)||'';
      let parsed={};
      try{
        const m=text.match(/\{[\s\S]*\}/);
        if(m) parsed=JSON.parse(m[0]);
      }catch(pe){}
      return ok({parsed:parsed,rawText:text});
    }

    // ── Gemini Vision OCR ─────────────────────
    if(d.action==='ocr_gemini'){
      const apiKey=PropertiesService.getScriptProperties().getProperty('GEMINI_API_KEY');
      if(!apiKey) return error('API 키가 설정되지 않았습니다. GAS 스크립트 속성에 GEMINI_API_KEY를 등록하세요.');
      const payload={
        contents:[{parts:[
          {inline_data:{mime_type:d.mediaType||'image/jpeg',data:d.imageBase64}},
          {text:'이미지에서 전도 보고서 정보를 추출해 JSON으로만 응답: {"location":"거주지역","name":"성씨","age":"나이","marriage":"기혼또는미혼","gender":"남자또는여자","meet_place":"만난장소","job":"직업","religion":"종교","target_type":"대상자구분","heaven":"있었고또는없었고","reason":"믿음또는행위","q10":"특별경험","q12":"기도제목"}'}
        ]}]
      };
      const resp=UrlFetchApp.fetch(
        'https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key='+apiKey,
        {method:'post',headers:{'Content-Type':'application/json'},payload:JSON.stringify(payload),muteHttpExceptions:true}
      );
      const res=JSON.parse(resp.getContentText());
      if(res.error) return error('Gemini API 오류: '+res.error.message);
      const text=((res.candidates||[])[0]||{}).content&&res.candidates[0].content.parts[0].text||'';
      const ocrText=text;
      let parsed={};
      try{
        const m=text.match(/\{[\s\S]*\}/);
        if(m) parsed=JSON.parse(m[0]);
      }catch(pe){}
      return ok({parsed:parsed,ocrText:ocrText});
    }

    const ss=SpreadsheetApp.openById(SHEET_ID);
    const s=ss.getSheetByName(SHEET_NAME)||ss.insertSheet(SHEET_NAME);
    if(s.getLastRow()===0) writeHeader(s);
    const savedAt=Utilities.formatDate(new Date(),'Asia/Seoul','yyyy-MM-dd HH:mm:ss');
    s.appendRow([
      savedAt,d.week,d.date,d.doc_number,d.evang_type,d.evang_count,
      d.jo,d.leader,d.trainer,d.trainee1,
      d.leader_step,d.trainer_step,d.trainee_step,
      d.contact,d.gospel,d.target,d.decision,d.reject,d.confirm,d.believer,
      d.location,d.name,d.age,d.marriage,d.gender,
      d.meet_place,d.gospel_place,d.target_type,d.relation_member,
      d.religion,d.job,d.heaven,d.reason,d.yeopjeop,d.diag_result,
      d.testimony6,d.testimony7,
      d.leader_part,d.leader_action,d.trainer_part,d.trainer_action,
      d.trainee_part,d.trainee_action,
      d.result,d.immed_nurture,d.followup,d.followup_reason,
      d.q10,d.q11,d.q12,d.q13
    ]);
    if(SEND_EMAIL) sendEmailAlert(d,savedAt);
    return ok({});
  }catch(err){return error(err.toString());}
}

// ── doGet : 기록 조회 ────────────────────────
function doGet(e){
  try{
    const action=e&&e.parameter&&e.parameter.action;
    if(action==='get_jo'){
      const raw=PropertiesService.getScriptProperties().getProperty('JO_DATA');
      return ok({joData:raw?JSON.parse(raw):[]});
    }
    // 문서번호로 단일 레코드 조회 (admin에서 수정 시 사용)
    if(action==='get_by_doc'){
      const docNumber=e.parameter.docNumber||'';
      const s=SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);
      if(!s||s.getLastRow()<2) return error('레코드 없음');
      const rows=s.getDataRange().getValues();
      const h=rows[0];
      const DATE_COLS=new Set(['저장일시','전도일']);
      const fmt=(v)=>{ if(v instanceof Date) return Utilities.formatDate(v,'Asia/Seoul','yyyy-MM-dd'); return v; };
      // 동일 문서번호 중 최신 레코드 반환
      let found=null;
      rows.slice(1).forEach((r,i)=>{
        const o={rowIndex:i+2};
        h.forEach((k,j)=>{ o[k]=DATE_COLS.has(k)?fmt(r[j]):r[j]; });
        if(String(o['문서번호'])===String(docNumber)){
          if(!found||(String(o['저장일시'])>String(found['저장일시']))) found=o;
        }
      });
      if(!found) return error('해당 문서번호를 찾을 수 없습니다: '+docNumber);
      return ok({record:found});
    }
    const s=SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);
    if(!s||s.getLastRow()<2) return ok({records:[]});
    const rows=s.getDataRange().getValues();
    const h=rows[0];
    const DATE_COLS=new Set(['저장일시','전도일']);
    const fmt=(v)=>{
      if(v instanceof Date) return Utilities.formatDate(v,'Asia/Seoul','yyyy-MM-dd');
      return v;
    };
    const rec=rows.slice(1).map((r,i)=>{
      const o={rowIndex:i+2};
      h.forEach((k,j)=>{ o[k]=DATE_COLS.has(k)?fmt(r[j]):r[j]; });
      return o;
    }).reverse();
    return ok({records:rec});
  }catch(err){return error(err.toString());}
}
